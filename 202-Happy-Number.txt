
////202. Happy Number
int getNext(int n){
        int sum=0;
        while(n>0){
            int digit=n%10;
            sum+=digit*digit;
            n/=10;
        }
        return sum;
    }
    bool isHappy(int n) {
       unordered_set<int> seen;///set for same numbers
       while(n!=1 && !seen.count(n)){
            seen.insert(n);
            n=getNext(n);
       }
       return n==1;
    }
