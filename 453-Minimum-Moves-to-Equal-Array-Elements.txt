//453. Minimum Moves to Equal Array Elements

int minMoves(vector<int>& nums) {
        int S = 0;
        int n = nums.size();
        int minVal = INT_MAX;

        for (int i = 0; i < n; i++) {
            S += nums[i];
            minVal = min(minVal, nums[i]);
        }

        return S - (minVal * n);
    }
